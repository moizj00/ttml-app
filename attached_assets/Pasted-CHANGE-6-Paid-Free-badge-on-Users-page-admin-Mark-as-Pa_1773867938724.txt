9b — New admin.repairLetterState mutation (server/routers.ts)
typescriptrepairLetterState: adminProcedure
  .input(z.object({ letterId: z.number() }))
  .mutation(async ({ input, ctx }) => {
    const [letter] = await db.select().from(letters)
      .where(eq(letters.id, input.letterId));
    if (!letter) throw new TRPCError({ code: "NOT_FOUND" });

    const [version] = await db.select().from(letterVersions)
      .where(eq(letterVersions.letterId, input.letterId))
      .limit(1);

    const [failedJob] = await db.select().from(pipelineJobs)
      .where(and(
        eq(pipelineJobs.letterId, input.letterId),
        eq(pipelineJobs.status, "failed")
      ))
      .limit(1);

    const findings: string[] = [];

    // Detect broken state: stuck in processing with a failed job and no content
    const isStuck =
      letter.status === "processing" && failedJob && !version;

    if (isStuck) {
      await db.update(letters)
        .set({ status: "submitted", updatedAt: new Date() })
        .where(eq(letters.id, input.letterId));

      findings.push(`Detected stuck-processing state. Reset to "submitted".`);
      findings.push(`Failed job ID: ${failedJob.id} (${failedJob.stage}).`);
      findings.push(`No content version found — letter can now be re-queued.`);
    } else {
      findings.push(`Letter #${input.letterId} is in "${letter.status}" — no repair needed.`);
      if (version)  findings.push(`Content version exists (v${version.version}).`);
      if (failedJob) findings.push(`Warning: failed job found (${failedJob.stage}) but letter has content.`);
    }

    await logAdminAction({
      adminId: ctx.user.id,
      action: "repair_letter_state",
      targetLetterId: input.letterId,
      metadata: { findings, wasStuck: isStuck },
    });

    return { repaired: isStuck, findings };
  }),
