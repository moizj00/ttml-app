def generate_lawyer_letter(user_request):
    # --- STAGE 1: RESEARCH ---
    # Goal: Get case law & facts
    try:
        context = Perplexity_Research(user_request) 
    except CreditsDepletedError:
        context = OpenAI_Search_Research(user_request) # Backup

    # --- STAGE 2: THE DRAFT ---
    # Goal: Build the legal skeleton
    try:
        initial_draft = Claude_Sonnet(context, "Draft a legal letter")
    except TokenLimitError:
        initial_draft = OpenAI_GPT4o(context, "Draft a legal letter") # Backup

    # --- STAGE 3: ASSEMBLY ---
    # Goal: Combine research citations into the draft
    assembled_doc = Assembly_Logic(initial_draft, context)

    # --- STAGE 4: THE FINAL POLISH (Priority Stage) ---
    # Goal: Perfect lawyer tone - ONLY use Claude unless absolutely empty
    try:
        final_letter = Claude_Opus(assembled_doc, "Refine into a finished lawyer letter")
    except TokenLimitError:
        final_letter = OpenAI_GPT4o(assembled_doc, "Refine into a finished lawyer letter")

    return final_letter
